CREATE TABLE `Activity` (`AcID` bigint(20) NOT NULL AUTO_INCREMENT, `Title` varchar(255) DEFAULT NULL, `CaID` int(11) DEFAULT NULL, `BuID` bigint(20) DEFAULT NULL, `Status` varchar(255) DEFAULT NULL, `Started` date DEFAULT NULL, `Ended` date DEFAULT NULL, `Internal` tinyint(1) DEFAULT `0`, PRIMARY KEY (`AcID`), FOREIGN KEY (`CaID`) REFERENCES `Campus` (`CaID`), FOREIGN KEY (`BuID`) REFERENCES `Budget` (`BuID`), ADD KEY `CaID` (`CaID`), ADD KEY `BuID` (`BuID`), ADD KEY `AcID` (`AcID`), ADD KEY `Title` (`Title`));
CREATE TABLE `Allocation` (`StaffID` bigint(20) DEFAULT NULL, `AcID` bigint(20) DEFAULT NULL, FOREIGN KEY (`StaffID`) REFERENCES `Staff` (`StaffID`), FOREIGN KEY (`AcID`) REFERENCES `Activity` (`AcID`));
CREATE TABLE `Budget` (`BuID` bigint(20) NOT NULL AUTO_INCREMENT, `Amount` bigint(20) DEFAULT NULL, `Approver` varchar(255) DEFAULT NULL, `Payee` varchar(255) DEFAULT NULL, `Status` varchar(255) DEFAULT NULL, ADD PRIMARY KEY (`BuID`), ADD KEY `BuID` (`BuID`));
CREATE TABLE `Campus` (`CaID` int(11) NOT NULL AUTO_INCREMENT, `Address` varchar(255) DEFAULT NULL, `GmName` varchar(255) DEFAULT NULL, `Country` varchar(255) DEFAULT NULL, `Status` varchar(255) DEFAULT NULL, ADD PRIMARY KEY (`CaID`), ADD KEY `CaID` (`CaID`));
CREATE TABLE `Classroom` (`RmID` bigint(20) NOT NULL AUTO_INCREMENT, `Capacity` bigint(20) DEFAULT NULL, `CaID` int(11) DEFAULT NULL, `Location` varchar(255) DEFAULT NULL, `Type` varchar(255) DEFAULT NULL, `Status` varchar(255) DEFAULT NULL, ADD PRIMARY KEY (`RmID`), ADD KEY `CaID` (`CaID`), ADD KEY `RmID` (`RmID`); FOREIGN KEY (`CaID`) REFERENCES `Campus` (`CaID`));
CREATE TABLE `Department` (`DeptID` bigint(20) NOT NULL AUTO_INCREMENT, `DeptName` varchar(255) DEFAULT NULL, `Faculty` varchar(255) DEFAULT NULL, `HODName` varchar(255) DEFAULT NULL, `Address` varchar(255) DEFAULT NULL, ADD PRIMARY KEY (`DeptID`), ADD KEY `DeptID` (`DeptID`));
CREATE TABLE `Enrolment` (`EID` bigint(20) NOT NULL AUTO_INCREMENT, `StudentID` bigint(20) DEFAULT NULL, `ModuleID` bigint(20) DEFAULT NULL, `Score` int(11) DEFAULT `1`, `Semester` varchar(255) DEFAULT NULL, `Year` bigint(20) DEFAULT NULL, ADD PRIMARY KEY (`EID`), ADD KEY `EID` (`EID`), ADD KEY `StudentID` (`StudentID`), ADD KEY `ModuleID` (`ModuleID`), ADD KEY `Year` (`Year`); FOREIGN KEY (`StudentID`) REFERENCES `Student` (`StudentID`), FOREIGN KEY (`ModuleID`) REFERENCES `Module` (`ModuleID`));
CREATE TABLE `Module` (`ModuleID` bigint(20) NOT NULL AUTO_INCREMENT, `ModuleName` varchar(255) DEFAULT NULL, `DeptID` bigint(20) DEFAULT NULL, `Programme` varchar(255) DEFAULT NULL, `TMode` varchar(255) DEFAULT NULL, `Date_Of_Firstoffer` date DEFAULT NULL, ADD PRIMARY KEY (`ModuleID`), ADD KEY `DeptID` (`DeptID`),ADD KEY `ModuleID` (`ModuleID`), ADD KEY `ModuleName` (`ModuleName`), FOREIGN KEY (`DeptID`) REFERENCES `Department` (`DeptID`));
CREATE TABLE `SpaceAssign` (`SpaID` bigint(20) NOT NULL AUTO_INCREMENT, `RmID` bigint(20) DEFAULT NULL, `TID` bigint(20) DEFAULT NULL, `Manager` varchar(255) DEFAULT NULL, `Approved` varchar(255) DEFAULT NULL, ADD PRIMARY KEY (`SpaID`), ADD KEY `RmID` (`RmID`), ADD KEY `TID` (`TID`), FOREIGN KEY (`RmID`) REFERENCES `Classroom` (`RmID`), FOREIGN KEY (`TID`) REFERENCES `Teach` (`TID`));
CREATE TABLE `Staff` (`StaffID` bigint(20) NOT NULL AUTO_INCREMENT, `Title` varchar(32) DEFAULT NULL, `FirstName` varchar(255) DEFAULT NULL, `LastName` varchar(255) DEFAULT NULL, `DeptID` bigint(20) DEFAULT NULL, `CaID` int(11) DEFAULT NULL, `Joined` date DEFAULT NULL, `LeftD` date DEFAULT NULL, `Current` tinyint(1) DEFAULT `1`, `Salary` bigint(20) DEFAULT NULL, `ContractType` varchar(32) DEFAULT NULL, `SupervisorID` bigint(20) DEFAULT NULL, ADD PRIMARY KEY (`StaffID`), ADD KEY `CaID` (`CaID`), ADD KEY `StaffID` (`StaffID`), ADD KEY `FirstName` (`FirstName`), ADD KEY `LastName` (`LastName`), ADD KEY `DeptID` (`DeptID`), ADD KEY `Current` (`Current`), ADD KEY `SupervisorID` (`SupervisorID`), FOREIGN KEY (`DeptID`) REFERENCES `Department` (`DeptID`), FOREIGN KEY (`CaID`) REFERENCES `Campus` (`CaID`));
CREATE TABLE `Student` (`StudentID` bigint(20) NOT NULL AUTO_INCREMENT, `StudentName` varchar(255) DEFAULT NULL, `DoB` date DEFAULT NULL, `Address` varchar(255) DEFAULT NULL, `StudentType` varchar(255) DEFAULT NULL, `Date_Of_Start` date DEFAULT NULL, ADD PRIMARY KEY (`StudentID`), ADD KEY `StudentID` (`StudentID`));
CREATE TABLE `Teach` (`TID` bigint(20) NOT NULL AUTO_INCREMENT, `ModuleID` bigint(20) DEFAULT NULL, `StaffID` bigint(20) DEFAULT NULL, `SpaID` bigint(20) DEFAULT NULL, `Semester` varchar(255) DEFAULT NULL, `Year` bigint(20) DEFAULT NULL, ADD PRIMARY KEY (`TID`), ADD KEY `SpaID` (`SpaID`), ADD KEY `TID` (`TID`), ADD KEY `ModuleID` (`ModuleID`), ADD KEY `StaffID` (`StaffID`), ADD KEY `Semester` (`Semester`), ADD KEY `Year` (`Year`) FOREIGN KEY (`StaffID`) REFERENCES `Staff` (`StaffID`), FOREIGN KEY (`SpaID`) REFERENCES `SpaceAssign` (`SpaID`), FOREIGN KEY (`ModuleID`) REFERENCES `Module` (`ModuleID`));
